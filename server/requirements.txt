#     python -m pip install -r server/requirements.txt

# timezones (e.g. Asia/Jerusalem) only resolve there with tzdata installed;
# without it the driver falls back to server-local time + MAINT_TZ_WARN. Not
# needed on Linux/macOS (system tz db is used).
